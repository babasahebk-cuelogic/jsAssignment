                            }

                            // Copy reference 
                            let newUser = user;
                            console.log(newUser.age);
                            
                            // Changed Reference value it will change real object also

                            // Constant object we can change the value or add the value
                            const userInfo = {
                                name: 'Babasaheb'
                            }
                            userInfo.age = 25;
                            console.log(`UserInfo object value:  ${userInfo.name} and age ${userInfo.age}`);
